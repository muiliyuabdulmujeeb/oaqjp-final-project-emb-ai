"""Server.py file to serve emotional detector over webpage"""

from flask import Flask, render_template, request
from EmotionDetection import emotion_detector


app = Flask("emotion_detection")

@app.route("/")
def entry():
    """serving static file and template to user"""
    return render_template("index.html")


@app.route("/emotionDetector")
def emotion_detector_route():
    """emotion detector function"""
    text_to_analyze = request.args.get("textToAnalyze", type=str)
    response = emotion_detector(text_to_analyze)

    if response['dominant_emotion'] is None:
        return "Invalid text! Please try again!"

    return f"""For the given statement, the system response is
    'anger':  {response['anger']}, 'disgust':  {response['disgust']},
    'fear':  {response['fear']}, 'joy':  {response['joy']}
    and 'sadness':  {response['sadness']}. 
    The dominant emotion is  {response['dominant_emotion']}"""

app.run(host="0.0.0.0", port=5000, debug=True)
